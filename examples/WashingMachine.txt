name WashingMachine
comment: Simulation of a washing machine cycle: filling, washing, rinsing, spinning, done.
Input:
Output: status

Variables:
    int tIdle   = 1000
    int tFill   = 3000
    int tWash   = 5000
    int tRinse  = 4000
    int tSpin   = 2000

States:
    state IDLE    [ return "IDLE: waiting before start"; ]
    state FILL    [ return "FILL: filling water"; ]
    state WASH    [ return "WASH: washing clothes"; ]
    state RINSE   [ return "RINSE: rinsing"; ]
    state SPIN    [ return "SPIN: spinning"; ]
    state DONE    [ return "DONE: cycle finished"; ]

Transitions:
    IDLE    --> FILL  : @ tIdle
    FILL    --> WASH  : @ tFill
    WASH    --> RINSE : @ tWash
    RINSE   --> SPIN  : @ tRinse
    SPIN    --> DONE  : @ tSpin
    DONE    --> IDLE  : @ 2000
