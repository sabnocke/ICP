name TOF5s
# Timer to off.
Input:
    in1
Output:
    out1
Variables:
    int timeout = 5000
States:
state IDLE [ output("out1", 0) ]
state ACTIVE [ output("out1", 1) ]
state TIMING [ ]
Transitions:
    IDLE --> ACTIVE: in1 [ tonumber(valueof("in")) == 1 ]
    ACTIVE --> TIMING: in1 [ tonumber(valueof("in")) == 0 ]
    TIMING --> ACTIVE : in1 [ tonumber(valueof("in")) == 1 ]
    TIMING --> IDLE : @ timeout
